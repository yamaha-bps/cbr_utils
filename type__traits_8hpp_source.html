23 /***************************************************************************
24  * \brief Returns type of variable as string_view
25  ***************************************************************************/
26 template<typename T>
27 constexpr auto type_name() noexcept
28 {
29  std::string_view name = "Error: unsupported compiler", prefix, suffix;
30 #ifdef __clang__
31  name = __PRETTY_FUNCTION__;
32  prefix = "auto cbr::type_name() [T = ";
33  suffix = "]";
34 #elif defined(__GNUC__)
35  name = __PRETTY_FUNCTION__;
36  prefix = "constexpr auto cbr::type_name() [with T = ";
37  suffix = "]";
38 #elif defined(_MSC_VER)
39  name = __FUNCSIG__;
40  prefix = "auto __cdecl cbr::type_name<";
41  suffix = ">(void) noexcept";
42 #endif
43  name.remove_prefix(prefix.size());
44  name.remove_suffix(suffix.size());
45  return name;
46 }
