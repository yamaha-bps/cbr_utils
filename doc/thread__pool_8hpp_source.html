66 // the constructor just launches some amount of workers
67 inline ThreadPool::ThreadPool(size_t threads) : stop(false)
68 {
69  for (size_t i = 0; i < threads; ++i) {
70  workers.emplace_back([this] {
71  for (;;) {
72  std::function<void()> task;
73 
74  {
75  std::unique_lock lock(this->queue_mutex);
76  this->condition.wait(lock, [this] { return this->stop || !this->tasks.empty(); });
77  if (this->stop && this->tasks.empty()) { return; }
78  task = std::move(this->tasks.front());
79  this->tasks.pop();
80  }
81 
82  task();
83  }
84  });
85  }
86 }

89 template<class F, class... Args>
90 auto ThreadPool::enqueue(F && f, Args &&... args)
91  -> std::future<typename std::result_of<F(Args...)>::type>
92 {
93  using return_type = typename std::result_of<F(Args...)>::type;
94 
95  auto task = std::make_shared<std::packaged_task<return_type()>>(
96  std::bind(std::forward<F>(f), std::forward<Args>(args)...));
97 
98  std::future<return_type> res = task->get_future();
99  {
100  std::unique_lock lock(queue_mutex);
101 
102  // don't allow enqueueing after stopping the pool
103  if (stop) { throw std::runtime_error("enqueue on stopped ThreadPool"); }
104 
105  tasks.emplace([task]() { (*task)(); });
106  }
107  condition.notify_one();
108  return res;
109 }

111 // the destructor joins all threads
112 inline ThreadPool::~ThreadPool()
113 {
114  {
115  std::unique_lock lock(queue_mutex);
116  stop = true;
117  }
118  condition.notify_all();
119  for (std::thread & worker : workers) { worker.join(); }
120 }
