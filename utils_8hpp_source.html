117 inline typename clock_t::time_point fromDateStr(const std::string & str)
118 {
119  std::istringstream iss{str};
120  std::tm tm{};
121 
122  if (!(iss >> std::get_time(&tm, "%Y-%m-%d_%H-%M-%S"))) {
123  throw std::invalid_argument("Date has wrong format");
124  }
125 
126  tm.tm_isdst = -1;
127 
128  const auto tOut = clock_t::from_time_t(std::mktime(&tm));
129  const auto pos = str.find('.');
130  std::chrono::nanoseconds nanoSec(0);
131  if (pos != std::string::npos) {
132  try {
133  nanoSec = std::chrono::duration_cast<std::chrono::nanoseconds>(
134  std::chrono::duration<double>(std::stod(str.substr(pos))));
135  } catch (const std::exception &) {}
136  }
137 
138  return tOut + nanoSec;
