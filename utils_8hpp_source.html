80 template<typename clock_t = std::chrono::system_clock>
81 inline typename clock_t::time_point fromDateStr(const std::string & str)
82 {
83  std::istringstream iss{str};
84  std::tm tm{};
85 
86  if (!(iss >> std::get_time(&tm, "%Y-%m-%d_%H-%M-%S"))) {
87  throw std::invalid_argument("Date has wrong format");
88  }
89 
90  tm.tm_isdst = -1;
91 
92  const auto tOut = clock_t::from_time_t(std::mktime(&tm));
93  const auto pos = str.find('.');
94  std::chrono::nanoseconds nanoSec(0);
95  if (pos != std::string::npos) {
96  try {
97  nanoSec = std::chrono::duration_cast<std::chrono::nanoseconds>(
98  std::chrono::duration<double>(std::stod(str.substr(pos))));
99  } catch (const std::exception &) {}
100  }
101 
102  return tOut + nanoSec;
103 }
