49  }));

51  } else if constexpr (::cbr::is_scoped_enum_v<T>) { // NOLINT
52  using UT = std::underlying_type_t<T>;

54  } else if constexpr (::cbr::is_std_tuple_v<T>) { // NOLINT

63  } else {

68  static bool decode(const Node & yaml, T & val)
69  {
70  if constexpr (boost::hana::Struct<T>::value) {
71  boost::hana::for_each(boost::hana::keys(val), [&](auto key) {
72  using ValT = std::decay_t<decltype(boost::hana::at_key(val, key))>;
73 
74  char const * key_c = boost::hana::to<char const *>(key);
75  if constexpr (::cbr::is_std_optional_v<ValT>) {
76  if (!yaml[key_c] || yaml[key_c].IsNull()) {
77  boost::hana::at_key(val, key) = std::nullopt;
78  } else {
79  boost::hana::at_key(val, key) = yaml[key_c].template as<typename ValT::value_type>();
80  }
81  } else {
82  boost::hana::at_key(val, key) = yaml[key_c].template as<ValT>();
83  }
84  });
85  return true;
86  } else if constexpr (::cbr::is_scoped_enum_v<T>) { // NOLINT
87  try {
88  using UT = std::underlying_type_t<T>;
89  const auto v = yaml.as<UT>();
90  if (v < static_cast<UT>(T::N_ELEMENTS)) {
91  val = static_cast<T>(v);
92  } else {
93  val = T::UNKOWN;
94  }
95  } catch (const std::exception & e) {
96  return false;
97  }
98  return true;
99  } else if constexpr (::cbr::is_std_tuple_v<T>) { // NOLINT
100  if (!yaml.IsSequence()) { return false; }
101  if (yaml.size() != std::tuple_size_v<T>) { return false; }
102 
103  ::cbr::static_for_index<std::tuple_size_v<T>>([&](auto i) {
104  std::get<i.value>(val) = yaml[i].template as<std::tuple_element_t<i.value, T>>();
105  });
106 
107  return true;
108 
109  } else if constexpr (::cbr::is_std_optional_v<T>) { // NOLINT
110  if (!yaml || yaml.IsNull()) {
111  val = std::nullopt;
112  return true;
113  }
114  try {
115  using UT = typename T::value_type;
116  val = yaml.as<UT>();
117  } catch (const std::exception & e) {
118  return false;
119  }
120  return true;
121  } else {
122  static_assert(::cbr::false_v<T>, "Unsupported type for YAML decoding.");
123  }
124  }
125 };
